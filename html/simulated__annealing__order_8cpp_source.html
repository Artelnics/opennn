206 SimulatedAnnealingOrder::SimulatedAnnealingOrderResults* SimulatedAnnealingOrder::perform_order_selection(void)
207 {
208  SimulatedAnnealingOrderResults* results = new SimulatedAnnealingOrderResults();
209 
210  NeuralNetwork* neural_network_pointer = training_strategy_pointer->get_performance_functional_pointer()->get_neural_network_pointer();
211  MultilayerPerceptron* multilayer_perceptron_pointer = neural_network_pointer->get_multilayer_perceptron_pointer();
212 
213  const size_t inputs_number = multilayer_perceptron_pointer->get_inputs_number();
214  const size_t outputs_number = multilayer_perceptron_pointer->get_outputs_number();
215 
216  size_t optimal_order, current_order;
217  Vector<double> optimum_performance(2);
218  Vector<double> current_order_performance(2);
219  Vector<double> optimum_parameters, current_parameters;
220 
221  Vector<double> history_row(2);
222  Vector<double> parameters_history_row;
223  double current_training_performance, current_generalization_performance;
224 
225  bool end = false;
226  size_t iterations = 0;
227  size_t generalization_failures = 0;
228  size_t random_failures = 0;
229  size_t upper_bound;
230  size_t lower_bound;
231 
232  time_t beginning_time, current_time;
233  double elapsed_time;
234 
235  double temperature;
236  double boltzmann_probability;
237  double random_uniform;
238 
239  if (display)
240  std::cout << "Performing order selection with simulated annealing method..." << std::endl;
241 
242  time(&beginning_time);
243 
244  optimal_order = (size_t)(minimum_order +
245  calculate_random_uniform(0.,1.)*(maximum_order - minimum_order));
246  optimum_performance = calculate_performances(optimal_order);
247  optimum_parameters = get_parameters_order(optimal_order);
248 
249  current_training_performance = optimum_performance[0];
250  current_generalization_performance = optimum_performance[1];
251 
252  temperature = current_generalization_performance;
253 
254  if (reserve_performance_data)
255  {
256  history_row[0] = optimal_order;
257  history_row[1] = current_training_performance;
258  results->performance_data.push_back(history_row);
259  }
260 
261  if (reserve_generalization_performance_data)
262  {
263  history_row[0] = optimal_order;
264  history_row[1] = current_generalization_performance;
265  results->generalization_performance_data.push_back(history_row);
266  }
267 
268  if (reserve_parameters_data)
269  {
270  parameters_history_row = get_parameters_order(optimal_order);
271  parameters_history_row.insert(parameters_history_row.begin(),optimal_order);
272  results->parameters_data.push_back(parameters_history_row);
273  }
274 
275  time(&current_time);
276  elapsed_time = difftime(current_time, beginning_time);
277 
278  if (display)
279  {
280  std::cout << "Initial values : " << std::endl;
281  std::cout << "Hidden perceptrons : " << optimal_order << std::endl;
282  std::cout << "Final Training Performance : " << optimum_performance[0] << std::endl;
283  std::cout << "Final generalization performance : " << optimum_performance[1] << std::endl;
284  std::cout << "Temperature : " << temperature << std::endl;
285  std::cout << "Elapsed time : " << elapsed_time << std::endl;
286  }
287 
288  while (!end){
289 
290  upper_bound = fmin(maximum_order, optimal_order + (maximum_order-minimum_order)/3);
291  if (optimal_order <= (maximum_order-minimum_order)/3)
292  lower_bound = minimum_order;
293  else
294  lower_bound = optimal_order - (maximum_order-minimum_order)/3;
295 
296  current_order = (size_t)(lower_bound + calculate_random_uniform(0.,1.)*(upper_bound - lower_bound));
297  while (current_order == optimal_order)
298  {
299  current_order = (size_t)(lower_bound + calculate_random_uniform(0.,1.)*(upper_bound - lower_bound));
300  random_failures++;
301  if (random_failures >= 5 && optimal_order != minimum_order)
302  current_order = optimal_order - 1;
303  else if (random_failures >= 5 && optimal_order != maximum_order)
304  current_order = optimal_order + 1;
305  }
306  random_failures = 0;
307 
308  current_order_performance = calculate_performances(current_order);
309  current_training_performance = current_order_performance[0];
310  current_generalization_performance = current_order_performance[1];
311  current_parameters = get_parameters_order(optimal_order);
312 
313  boltzmann_probability = fmin(1, exp(-(current_generalization_performance-optimum_performance[1])/temperature));
314  random_uniform = calculate_random_uniform(0.,1.);
315 
316  if ((boltzmann_probability <= random_uniform)
317  || (fabs(optimum_performance[1]-current_generalization_performance) <= tolerance
318  && current_order >= optimal_order))
319  // Generalization failures
320  {
321  generalization_failures++;
322  }else
323  // Generalization success
324  {
325  optimal_order = current_order;
326  optimum_performance = current_order_performance;
327  optimum_parameters = get_parameters_order(optimal_order);
328  }
329 
330  time(&current_time);
331  elapsed_time = difftime(current_time, beginning_time);
332 
333  if (reserve_performance_data)
334  {
335  history_row[0] = current_order;
336  history_row[1] = current_order_performance[0];
337  results->performance_data.push_back(history_row);
338  }
339 
340  if (reserve_generalization_performance_data)
341  {
342  history_row[0] = current_order;
343  history_row[1] = current_order_performance[1];
344  results->generalization_performance_data.push_back(history_row);
345  }
346 
347  if (reserve_parameters_data)
348  {
349  parameters_history_row = get_parameters_order(current_order);
350  parameters_history_row.insert(parameters_history_row.begin(),current_order);
351  results->parameters_data.push_back(parameters_history_row);
352  }
353 
354  temperature = cooling_rate*temperature;
355 
356  iterations++;
357 
358  // Stopping criteria
359 
360  if (temperature < minimum_temperature)
361  {
362  end = true;
363  if (display)
364  std::cout << "Minimum temperature reached." << std::endl;
365  results->stopping_condition = SimulatedAnnealingOrder::MinimumTemperature;
366  }else if (elapsed_time > maximum_time)
367  {
368  end = true;
369  if (display)
370  std::cout << "Maximum time reached." << std::endl;
371  results->stopping_condition = SimulatedAnnealingOrder::MaximumTime;
372  }else if (optimum_performance[1] < generalization_performance_goal)
373  {
374  end = true;
375  if (display)
376  std::cout << "Generalization performance reached." << std::endl;
377  results->stopping_condition = SimulatedAnnealingOrder::GeneralizationPerformanceGoal;
378  }else if (generalization_failures >= maximum_generalization_failures)
379  {
380  end = true;
381  if (display)
382  std::cout << "Maximum generalization performance failures("<<generalization_failures<<") reached." << std::endl;
383  results->stopping_condition = SimulatedAnnealingOrder::MaximumGeneralizationFailures;
384  }else if (iterations >= maximum_iterations_number)
385  {
386  end = true;
387  if (display)
388  std::cout << "Maximum number of iterations reached." << std::endl;
389  results->stopping_condition = SimulatedAnnealingOrder::MaximumIterations;
390  }
391 
392  if (display)
393  {
394  std::cout << "Iteration : " << iterations << std::endl;
395  std::cout << "Hidden perceptron number : " << optimal_order << std::endl;
396  std::cout << "Final Training Performance : " << optimum_performance[0] << std::endl;
397  std::cout << "Final generalization performance : " << optimum_performance[1] << std::endl;
398  std::cout << "Current temperature : " << temperature << std::endl;
399  std::cout << "Elapsed time : " << elapsed_time << std::endl;
400  }
401 
402  }
403 
404  if (display)
405  std::cout << "Optimal order : " << optimal_order << std:: endl;
406 
407  multilayer_perceptron_pointer->set(inputs_number, optimal_order, outputs_number);
408  multilayer_perceptron_pointer->set_parameters(optimum_parameters);
409 
410  if (reserve_minimal_parameters)
411  results->minimal_parameters = optimum_parameters;
412 
413  results->optimal_order = optimal_order;
414  results->final_performance = optimum_performance[0];
415  results->final_generalization_performance = optimum_performance[1];
416  results->elapsed_time = elapsed_time;
417  results->iterations_number = iterations;
418 
419  return(results);
420 }
