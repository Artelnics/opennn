745 Matrix<size_t> TestingAnalysis::calculate_confusion_binary_classification(const Matrix<double>& target_data, const Matrix<double>& output_data, const double& decision_threshold) const

764 
765  // False negative
766

769  false_negative++;
770  }
771 
772  // False positive
773

776  false_positive++;
777  }
778 
779  // True negative
780 
781  if(target_data(i,0) < decision_threshold && output_data(i,0) < decision_threshold)
782  {
783  true_negative++;
784  }
785  }
786 
787  confusion(0,0) = true_positive;
788  confusion(0,1) = false_negative;
789  confusion(1,0) = false_positive;
790  confusion(1,1) = true_negative;
